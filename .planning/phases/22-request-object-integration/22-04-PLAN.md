    ```elixir
    defmodule Lockspire.Protocol.RequestObject do
      @moduledoc """
      Orchestrates JAR (RFC 9101) request-object consumption for `/authorize` and `/par`.

      Composes `Lockspire.Protocol.Jar.{decode/1, verify_signature/2, validate_claims/2}`
      into a single pipeline step that:

      1. Rejects outer-param conflicts (D-04) and `request` / `request_uri` collisions (D-06).
      2. Asserts the client has inline `jwks` registered (D-08).
      3. Decodes, verifies the signature (incl. RFC 9101 §10.8 typ-header check via
         `Jar.verify_signature/2`'s `:invalid_typ` branch — Plan 22-01), and validates
         the request JWT's claims with the configured `:max_age` ceiling (D-13).
      4. Projects JAR claims into the same flat-params shape `pushed_request_to_params/1`
         produces in `Lockspire.Protocol.AuthorizationRequest`, so `validate_with_client/3`
         runs unchanged.

      ## Out of scope (v1.4)

      - **JAR-by-reference (request_uri pointing to an external JWT URL).** Lockspire
        keeps `request_uri` semantics as Lockspire-issued PAR references only — see
        `validate_lockspire_request_uri/1` in `Lockspire.Protocol.AuthorizationRequest`.
      - **JAR decryption (RFC 9101 §6 nested JWE).** Deferred indefinitely (`JAR-04`).
      - **Client `jwks_uri` HTTP fetch.** Deferred (Phase 23+); requires HTTP cache,
        retry, and operator-visible failure surface.
      - **JAR substituting as client authentication at /par.** RFC 9126 §2.1's optional
        substitution path is NOT adopted in v1.4 — `ClientAuth.authenticate/3` continues
        to run independently and unchanged at `/par` (D-10).
      - **JTI replay cache.** v1.4 does NOT add a JTI cache. Replay protection is bounded
        only by the configurable `:jar_max_age_seconds` ceiling (default 600s, D-13).
        Lower the ceiling to reduce replay risk.

      ## Sealed-envelope posture (D-04, D-07)

      When `params["request"]` is present, the only outer params honored are
      `client_id` (REQUIRED for client lookup) and `request` itself. Every other
      outer param — including `redirect_uri`, `scope`, `state`, etc. — is rejected as
      `:request_object_conflict`. RFC 9101 §6.1 permits the stricter posture; §10.2
      recommends it. This eliminates the "outer-params-override-the-signed-request"
      smuggling vector.
      """

      alias Lockspire.Config
      alias Lockspire.Domain.Client
      alias Lockspire.Protocol.AuthorizationRequest.Error
      alias Lockspire.Protocol.Jar

      @type result ::
              {:ok, map()}
              | {:browser_error, Error.t()}
              | {:redirect_error, Error.t()}

      @doc """
      Consumes a JAR-by-value request, returning either the projected flat-params map
      (success) or a classified `Error` struct.

      Returns:
      - `{:ok, projected_params}` — JAR validated; downstream `validate_with_client/3`
        will run on `projected_params` unchanged.
      - `{:browser_error, %Error{}}` — JAR rejected; render first-party error page.
        This is the dominant path for JAR-validation failures because D-04 forbids the
        outer `redirect_uri` that would otherwise enable redirect-safe error reporting.
      - `{:redirect_error, %Error{}}` — reserved for the future case where outer
        `redirect_uri` is loosened. In v1.4 this branch is structurally unreachable
        but kept for symmetry with `Lockspire.Protocol.AuthorizationRequest`'s
        existing classifier shape (per D-16).
      """
      @spec consume(map(), Client.t(), keyword()) :: result()
      def consume(params, %Client{} = client, opts \\ []) when is_map(params) and is_list(opts) do
        with :ok <- reject_request_uri_collision(params),
             :ok <- reject_outer_param_conflicts(params),
             {:ok, jwt} <- fetch_request(params),
             :ok <- require_client_jwks(client),
             {:ok, %Jar{} = jar} <- decode_and_verify(jwt, client),
             :ok <- validate(jar, client, opts),
             {:ok, projected} <- project_to_params(jar, client) do
          {:ok, projected}
        end
      end

      # ---------------------------------------------------------------------------
      # Step 1 — D-06 mutual-exclusion of `request` and `request_uri`
      # ---------------------------------------------------------------------------

      defp reject_request_uri_collision(%{"request_uri" => uri}) when is_binary(uri) and uri != "" do
        {:browser_error,
         browser_error(
           :invalid_request,
           "request and request_uri cannot both be supplied",
           :request_object_and_request_uri_conflict
         )}
      end

      defp reject_request_uri_collision(_params), do: :ok

      # ---------------------------------------------------------------------------
      # Step 2 — D-04 sealed-envelope conflict rejection
      # Only "client_id" and "request" are permitted as outer params when a request
      # object is present. Symmetric with reject_request_uri_conflicts/1 in
      # AuthorizationRequest (authorization_request.ex:393-411).
      # ---------------------------------------------------------------------------

      @allowed_outer_keys ~w(client_id request)

      defp reject_outer_param_conflicts(params) do
        conflict_keys =
          params
          |> Enum.reject(fn {key, _value} -> key in @allowed_outer_keys end)
          |> Enum.filter(fn {_key, value} -> present?(value) end)

        case conflict_keys do
          [] ->
            :ok

          _ ->
            {:browser_error,
             browser_error(
               :invalid_request,
               "request cannot be combined with raw authorization parameters",
               :request_object_conflict
             )}
        end
      end

      # ---------------------------------------------------------------------------
      # Step 3 — extract the JWT
      # ---------------------------------------------------------------------------

      defp fetch_request(%{"request" => req}) when is_binary(req) and req != "" do
        {:ok, req}
      end

      defp fetch_request(_params) do
        # This path is unreachable in normal use because the splice in
        # AuthorizationRequest.validate/1 only routes to consume/3 when
        # params["request"] is non-empty. Defensive guard so consume/3 contract
        # is internally consistent.
        {:browser_error,
         browser_error(:invalid_request, "request parameter is required", :missing_request)}
      end

      # ---------------------------------------------------------------------------
      # Step 4 — D-08 require inline jwks
      # Disambiguates "client has no jwks at all" (operator UX: register a JWK)
      # from "client has malformed jwks" (operator UX: debug the JWK). The latter
      # would surface from Jar.verify_signature/2 as :invalid_client_keys after
      # this pre-check passes — we map that to :invalid_request_object_signature
      # to avoid leaking malformedness signal (Pitfall 3).
      # ---------------------------------------------------------------------------

      defp require_client_jwks(%Client{jwks: jwks}) when is_map(jwks) and map_size(jwks) > 0 do
        :ok
      end

      defp require_client_jwks(%Client{}) do
        {:browser_error,
         browser_error(
           :invalid_request_object,
           "Client has no registered jwks for request object signature verification",
           :client_jwks_missing
         )}
      end

      # ---------------------------------------------------------------------------
      # Step 5 — decode + verify_signature (handles WR-01 typ check via Jar primitive)
      # ---------------------------------------------------------------------------

      defp decode_and_verify(jwt, %Client{} = client) do
        with {:ok, %Jar{}} <- decode_step(jwt),
             {:ok, %Jar{} = verified_jar} <- verify_step(jwt, client) do
          {:ok, verified_jar}
        end
      end

      defp decode_step(jwt) do
        case Jar.decode(jwt) do
          {:ok, %Jar{} = jar} ->
            {:ok, jar}

          {:error, :invalid_jwt} ->
            {:browser_error,
             browser_error(
               :invalid_request_object,
               "Request object is not a valid JWT",
               :invalid_request_object_jwt
             )}
        end
      end

      defp verify_step(jwt, client) do
        case Jar.verify_signature(jwt, client) do
          {:ok, %Jar{} = jar} ->
            {:ok, jar}

          {:error, :invalid_typ} ->
            {:browser_error,
             browser_error(
               :invalid_request_object,
               "Request object has an unsupported typ header",
               :invalid_request_object_typ
             )}

          {:error, reason} when reason in [:invalid_signature, :no_matching_key, :invalid_client_keys] ->
            # :invalid_client_keys here means the jwks passed the require_client_jwks
            # nil-check but is malformed (e.g. valid map shape, invalid JWK fields).
            # Map to signature-failure to avoid leaking jwks-malformedness signal.
            {:browser_error,
             browser_error(
               :invalid_request_object,
               "Request object signature could not be verified",
               :invalid_request_object_signature
             )}
        end
      end

      # ---------------------------------------------------------------------------
      # Step 6 — claims validation (issuer, audience, expiration, max-age, nbf, iat)
      # ---------------------------------------------------------------------------

      defp validate(%Jar{} = jar, %Client{client_id: cid}, opts) do
        merged_opts =
          Keyword.merge(default_opts(), opts)
          |> Keyword.put(:expected_client_id, cid)

        case Jar.validate_claims(jar, merged_opts) do
          :ok ->
            :ok

          {:error, reason} when reason in [:missing_issuer, :invalid_issuer] ->
            {:browser_error,
             browser_error(
               :invalid_request_object,
               "Request object iss claim is missing or does not match client_id",
               :invalid_request_object_iss
             )}

          {:error, reason} when reason in [:missing_audience, :invalid_audience] ->
            {:browser_error,
             browser_error(
               :invalid_request_object,
               "Request object aud claim is missing or does not match the expected audience",
               :invalid_request_object_aud
             )}

          {:error, reason} when reason in [:missing_expiration, :invalid_expiration, :expired_token] ->
            {:browser_error,
             browser_error(
               :invalid_request_object,
               "Request object exp claim is missing, malformed, or in the past",
               :invalid_request_object_expired
             )}

          {:error, :expiration_too_far} ->
            {:browser_error,
             browser_error(
               :invalid_request_object,
               "Request object exp exceeds the configured maximum age",
               :invalid_request_object_max_age
             )}

          {:error, _other} ->
            # Catches :invalid_not_before, :invalid_issued_at, :invalid_claims_options.
            # :invalid_claims_options indicates an orchestrator programming error
            # (e.g. failure to pass expected_audience); should never reach the client
            # in normal use, but mapped through the same atom for fail-safe behavior.
            {:browser_error,
             browser_error(
               :invalid_request_object,
               "Request object failed claims validation",
               :invalid_request_object_claims
             )}
        end
      end

      defp default_opts do
        [
          expected_audience: Config.issuer!(),
          max_age: Config.jar_max_age_seconds(),
          leeway: 5
        ]
      end

      # ---------------------------------------------------------------------------
      # Step 7 — D-07 sealed-envelope projection from JAR claims into flat-params
      # Mirrors Lockspire.Protocol.AuthorizationRequest.pushed_request_to_params/1
      # (authorization_request.ex:452-466). NEVER reads from outer params (other
      # than client_id, which we use only for the iss cross-check above).
      # ---------------------------------------------------------------------------

      defp project_to_params(%Jar{claims: claims}, %Client{client_id: cid}) do
        projected =
          %{
            "client_id" => cid,
            "redirect_uri" => claims["redirect_uri"],
            "response_type" => claims["response_type"],
            "scope" => claims["scope"],
            "prompt" => claims["prompt"],
            "nonce" => claims["nonce"],
            "state" => claims["state"],
            "code_challenge" => claims["code_challenge"],
            "code_challenge_method" => claims["code_challenge_method"]
          }
          |> Enum.reject(fn {_k, v} -> is_nil(v) or v == "" end)
          |> Map.new()

        {:ok, projected}
      end

      # ---------------------------------------------------------------------------
      # Helpers
      # ---------------------------------------------------------------------------

      defp browser_error(error, description, reason_code) do
        %Error{
          error: to_string(error),
          error_description: description,
          reason_code: reason_code,
          redirect_uri: nil,
          state: nil
        }
      end
    ```
