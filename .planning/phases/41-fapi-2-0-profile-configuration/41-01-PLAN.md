```elixir
defmodule Lockspire.Protocol.SecurityProfile do
  alias Lockspire.Domain.ServerPolicy
  @type mode :: :inherit | :fapi_2_0_security | :none

  defmodule Resolved do
    @type t :: %__MODULE__{
            global_profile: ServerPolicy.security_profile(),  # :none | :fapi_2_0_security
            client_profile: Lockspire.Protocol.SecurityProfile.mode(),  # :inherit | :fapi_2_0_security | :none
            effective_profile: ServerPolicy.security_profile(),
            fapi_2_0_security?: boolean()
          }
    defstruct global_profile: :none,
              client_profile: :inherit,
              effective_profile: :none,
              fapi_2_0_security?: false
  end

  @spec resolve_effective_profile(ServerPolicy.t(), struct() | map() | nil) :: Resolved.t()
  def resolve_effective_profile(%ServerPolicy{} = server_policy, client)

  @spec allowed_signing_algorithms(ServerPolicy.security_profile()) :: [String.t()]
  # :fapi_2_0_security -> ["ES256", "PS256", "EdDSA"]
  # :none              -> ["RS256", "ES256", "PS256", "EdDSA"]
end
```

Resolution rules (verify these in tests):
- `client_profile: :inherit`            -> effective = global
- `client_profile: :fapi_2_0_security`  -> effective = :fapi_2_0_security (always)
- `client_profile: :none`               -> effective = :none (mixed-mode escape hatch — D-01 rationale)
- `client = nil`                        -> client_profile normalises to :inherit
- `fapi_2_0_security?` is true iff effective_profile == :fapi_2_0_security
