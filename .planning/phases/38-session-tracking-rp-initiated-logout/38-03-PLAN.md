```elixir
defmodule Lockspire.Web.EndSessionController do
  @moduledoc """
  Thin `/end_session` delivery adapter for OIDC RP-Initiated Logout 1.0.
  All validation logic lives in `Lockspire.Protocol.EndSession`.
  """

  use Phoenix.Controller, formats: [:html]

  require Logger

  alias Lockspire.Config
  alias Lockspire.Protocol.EndSession
  alias Lockspire.Storage.Ecto.Repository

  def show(conn, params), do: handle_end_session(conn, params)
  def create(conn, params), do: handle_end_session(conn, params)

  defp handle_end_session(conn, params) do
    request = %{params: params, opts: [client_store: Repository, key_store: Repository]}
    case EndSession.validate(request) do
      {:ok, %EndSession.Result{} = result} ->
        redirect_to_host_logout(conn, result)
      {:error, %EndSession.Error{} = error} ->
        conn
        |> put_status(error.status)
        |> put_resp_content_type("text/html")
        |> send_resp(error.status, "#{error.error}: #{error.error_description}")
    end
  end

  defp redirect_to_host_logout(conn, %EndSession.Result{} = result) do
    resolver = Lockspire.account_resolver!()

    return_to_payload = %{
      sid: result.sid,
      post_logout_redirect_uri: result.post_logout_redirect_uri,
      state: result.state
    }
    return_to_token = Phoenix.Token.sign(conn, "lockspire_logout", return_to_payload, max_age: 600)

    completion_url =
      Config.mount_path() <> "/end_session/complete"
      |> append_query_param("token", return_to_token)

    context = %{account_id: result.account_id, return_to: completion_url}

    # Guard: redirect_for_logout is @optional_callbacks — fall back to Config.logout_path()
    # directly with return_to param if the host has not implemented the callback.
    if function_exported?(resolver, :redirect_for_logout, 2) do
      interaction_result = resolver.redirect_for_logout(conn, context)

      destination =
        interaction_result.login_path
        |> append_query_param("return_to", interaction_result.return_to)
        |> append_query_params(interaction_result.params || %{})

      redirect(conn, to: destination)
    else
      destination = append_query_param(Config.logout_path(), "return_to", completion_url)
      redirect(conn, to: destination)
    end
  end

  def complete(conn, %{"token" => token}) do
    case Phoenix.Token.verify(conn, "lockspire_logout", token, max_age: 600) do
      {:ok, %{sid: sid, post_logout_redirect_uri: post_logout_redirect_uri, state: state}} ->
        _ = Repository.revoke_by_sid(sid)
        redirect_or_render_logged_out(conn, post_logout_redirect_uri, state)
      {:error, reason} ->
        Logger.warning("EndSession complete: invalid return_to token — #{inspect(reason)} — treating as logout success per D-10")
        redirect_or_render_logged_out(conn, nil, nil)
    end
  end

  def complete(conn, _params) do
    # Missing token param — treat as logout success (D-10)
    redirect_or_render_logged_out(conn, nil, nil)
  end

  defp redirect_or_render_logged_out(conn, nil, _state) do
    conn
    |> put_status(:ok)
    |> put_view(Lockspire.Web.EndSessionHTML)
    |> put_layout(false)
    |> render(:logged_out)
  end

  defp redirect_or_render_logged_out(conn, post_logout_redirect_uri, state)
       when is_binary(post_logout_redirect_uri) and post_logout_redirect_uri != "" do
    destination = append_query_param(post_logout_redirect_uri, "state", state)
    redirect(conn, external: destination)
  end

  # Query param helpers — mirrors authorize_controller.ex pattern
  defp append_query_params(path, params) when is_map(params) do
    Enum.reduce(params, path, fn {key, value}, acc ->
      append_query_param(acc, key, value)
    end)
  end

  defp append_query_param(path, _key, nil), do: path
  defp append_query_param(path, _key, ""), do: path

  defp append_query_param(path, key, value) do
    separator = if String.contains?(path, "?"), do: "&", else: "?"
    path <> separator <> URI.encode_query(%{to_string(key) => value})
  end
end
```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| RP browser → GET/POST /end_session | Untrusted input: id_token_hint, client_id, post_logout_redirect_uri, state |
| Host → GET /end_session/complete | Signed return_to token from Lockspire — verified before acting |
| EndSessionController → EndSessionProtocol | Controller is thin delivery; all trust decisions in protocol module |
| EndSessionController → Host AccountResolver | redirect_for_logout is optional callback; nil-safe fallback required |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-38-05 | Spoofing | open redirect via unregistered post_logout_redirect_uri | mitigate | `EndSession.validate_post_logout_redirect_uri/2` performs exact-match against `client.post_logout_redirect_uris` (D-15). Unregistered URI → 400, no redirect issued |
| T-38-06 | Tampering | forged id_token_hint from different OP | mitigate | `JOSE.JWT.verify_strict/3` with Lockspire's own signing keys; unsigned or wrong-key JWTs rejected (D-14). No exp check per OIDC spec. |
| T-38-07 | Tampering | cross-client logout via client_id/aud mismatch | mitigate | `validate_aud_if_client_id/3` rejects if client_id not in aud claim (D-20). Reason code: `:client_id_not_in_aud` |
| T-38-08 | Repudiation | return_to token replay | mitigate | `Phoenix.Token` max_age: 600 (10 min TTL). Stale tokens rejected; D-10 ensures user not stranded on rejection |
| T-38-09 | Spoofing | forged return_to token redirecting to attacker | mitigate | Phoenix.Token signing with conn secret_key_base; unsigned or tampered tokens fail verification; completion falls back to logged-out page (D-10) |
| T-38-10 | Denial of Service | stranded user on completion failure | mitigate | D-10: all failures in complete/2 render logged-out page. No error loop possible. |
| T-38-11 | Elevation | revoke_by_sid with arbitrary sid from URL param | mitigate | sid comes from verified Phoenix.Token payload (not raw URL param); verification precedes revocation (D-11) |
</threat_model>
