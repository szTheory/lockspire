```elixir
defmodule Lockspire.RAR.Fingerprint do
  @moduledoc """
  RFC 8785 (JCS) canonicalization + SHA-256 hashing of normalized
  authorization_details, used for ConsentGrant reuse-policy lookup.

  See `Lockspire.Protocol.ConsentPolicy.reusable_grant/4`.

  ## Determinism

  This implementation uses the `jcs` Hex package (v0.2.x) to canonicalize
  the JSON representation of `authorization_details` per RFC 8785 before
  hashing. This avoids the `Jason.encode!/1` map-iteration-order footgun
  documented in RESEARCH.md §"Pitfall 1".

  ## OTP-version sensitivity (Assumption A2)

  Float encoding under `:erlang.float_to_binary/2` may differ across OTP
  releases. Lockspire pins OTP via the Elixir floor (`~> 1.18` ⇒ OTP 25+).
  Validators SHOULD coerce numeric RAR values to integers or strings; CI
  MUST run on a single OTP minor per environment to avoid hash drift.
  """

  @doc """
  Computes a 32-byte SHA-256 fingerprint of the JCS canonicalization of the
  given `authorization_details` list.

  Returns `nil` for an empty list (RAR absent ⇒ no fingerprint binding).
  """
  @spec compute([map()]) :: binary() | nil
  def compute([]), do: nil

  def compute(authorization_details) when is_list(authorization_details) do
    authorization_details
    |> Jcs.encode()
    |> then(&:crypto.hash(:sha256, &1))
  end
end
```

```elixir
defmodule Lockspire.RAR.FingerprintTest do
  use ExUnit.Case, async: true

  alias Lockspire.RAR.Fingerprint

  describe "compute/1" do
    test "compute([]) returns nil (RAR absent)" do
      assert Fingerprint.compute([]) == nil
    end

    test "compute non-empty list returns 32-byte binary" do
      hash = Fingerprint.compute([%{"type" => "payment_initiation", "amount" => 100}])
      assert is_binary(hash)
      assert byte_size(hash) == 32
    end

    test "compute is invariant under map-key construction order (smoke test)" do
      a = [%{"type" => "x", "a" => 1, "b" => 2, "c" => 3}]
      b = [%{"type" => "x", "c" => 3, "b" => 2, "a" => 1}]
      assert Fingerprint.compute(a) == Fingerprint.compute(b)
    end

    test "compute differs on different values (collision sanity)" do
      assert Fingerprint.compute([%{"a" => 1}]) != Fingerprint.compute([%{"a" => 2}])
    end

    test "compute respects list order (arrays NOT canonicalized per RFC 8785 §3.2)" do
      d1 = %{"type" => "x", "actions" => ["read"]}
      d2 = %{"type" => "y", "actions" => ["write"]}
      assert Fingerprint.compute([d1, d2]) != Fingerprint.compute([d2, d1])
    end

    test "compute handles nested maps (3-level invariance)" do
      a = [%{"type" => "x", "creditor" => %{"name" => "Alice", "iban" => "DE..."}}]
      b = [%{"type" => "x", "creditor" => %{"iban" => "DE...", "name" => "Alice"}}]
      assert Fingerprint.compute(a) == Fingerprint.compute(b)
    end
  end
end
```

```elixir
defmodule Lockspire.RAR.FingerprintPropertyTest do
  use ExUnit.Case, async: true
  use ExUnitProperties

  alias Lockspire.RAR.Fingerprint

  property "fingerprint is invariant under map-key construction order" do
    check all keys <-
                StreamData.list_of(
                  StreamData.string(:alphanumeric, min_length: 1, max_length: 8),
                  min_length: 1,
                  max_length: 6
                ),
              values <- StreamData.list_of(StreamData.integer(), length: length(keys)),
              max_runs: 100 do
      pairs = Enum.zip(keys, values)
      a = pairs |> Enum.shuffle() |> Map.new() |> then(&[Map.put(&1, "type", "test")])
      b = pairs |> Enum.shuffle() |> Map.new() |> then(&[Map.put(&1, "type", "test")])

      assert Fingerprint.compute(a) == Fingerprint.compute(b)
    end
  end

  property "fingerprint is invariant under nested-map key reordering" do
    check all outer_keys <-
                StreamData.list_of(StreamData.string(:alphanumeric, min_length: 1, max_length: 6),
                  min_length: 1, max_length: 4),
              inner_keys <-
                StreamData.list_of(StreamData.string(:alphanumeric, min_length: 1, max_length: 6),
                  min_length: 1, max_length: 4),
              max_runs: 50 do
      inner_pairs = Enum.with_index(inner_keys) |> Enum.map(fn {k, i} -> {k, i} end)
      outer_pairs =
        Enum.with_index(outer_keys)
        |> Enum.map(fn {k, _} -> {k, inner_pairs |> Enum.shuffle() |> Map.new()} end)

      a = [%{"type" => "test"} |> Map.merge(outer_pairs |> Enum.shuffle() |> Map.new())]

      outer_pairs_b =
        Enum.map(outer_pairs, fn {k, _v} -> {k, inner_pairs |> Enum.shuffle() |> Map.new()} end)

      b = [%{"type" => "test"} |> Map.merge(outer_pairs_b |> Enum.shuffle() |> Map.new())]

      assert Fingerprint.compute(a) == Fingerprint.compute(b)
    end
  end
end
```

```elixir
defmodule Lockspire.RAR do
  @moduledoc """
  Public helpers for host RAR validator implementations.

  ## When to use

  Hosts implementing `Lockspire.Host.RarTypeValidator` SHOULD return
  `{:error, %Ecto.Changeset{}}` on validation failure and let the protocol
  layer call `error_description/1` to format the changeset into an
  RFC 9396 §6-compliant `error_description` string.

  Returning `{:error, binary}` is supported for hosts that prefer
  hand-rolled error messages.
  """

  @doc """
  Formats an `Ecto.Changeset` validation error into a single
  RFC 9396 §6-compliant `error_description` string.

  Strings are passed through unchanged.

  ## Examples

      iex> cs = {%{}, %{name: :string}}
      ...>      |> Ecto.Changeset.cast(%{}, [:name])
      ...>      |> Ecto.Changeset.validate_required([:name])
      iex> Lockspire.RAR.error_description(cs)
      "name: can't be blank"

      iex> Lockspire.RAR.error_description("custom message")
      "custom message"
  """
  @spec error_description(Ecto.Changeset.t() | String.t()) :: String.t()
  def error_description(%Ecto.Changeset{} = cs) do
    cs
    |> Ecto.Changeset.traverse_errors(fn {msg, opts} ->
      Enum.reduce(opts, msg, fn {key, value}, acc ->
        String.replace(acc, "%{#{key}}", to_string(value))
      end)
    end)
    |> Enum.map(fn {field, msgs} -> "#{field}: #{Enum.join(msgs, ", ")}" end)
    |> Enum.join("; ")
  end

  def error_description(description) when is_binary(description), do: description
end
```
