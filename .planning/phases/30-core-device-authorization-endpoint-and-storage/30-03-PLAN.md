Call `Lockspire.Protocol.DeviceAuthorization.authorize(%{params: params, authorization: List.first(get_req_header(conn, "authorization")), opts: [device_code_store: Lockspire.Storage.Ecto.Repository]})`.
    
    Analog reference: `lib/lockspire/web/controllers/pushed_authorization_request_controller.ex`
    Controller Pattern:
    ```elixir
      def create(conn, params) do
        authorization = List.first(get_req_header(conn, "authorization"))
    
        case DeviceAuthorization.authorize(%{
               params: params,
               authorization: authorization,
               opts: [client_store: Repository, device_code_store: Repository]
             }) do
          {:ok, %Success{} = success} ->
            conn
            |> put_cache_headers()
            |> put_status(:ok) # NOTE: For Device Auth it's usually 200 OK
            |> json(DeviceAuthorizationJSON.success_response(success))
    
          {:error, %Error{} = error} ->
            conn
            |> put_cache_headers()
            |> maybe_put_www_authenticate(error)
            |> put_status(error.status)
            |> json(DeviceAuthorizationJSON.error_response(error))
        end
      end
    ```

    Apply `put_cache_headers(conn)` setting `cache-control` to `no-store` and `pragma` to `no-cache`.
    Cache Headers Pattern:
    ```elixir
      defp put_cache_headers(conn) do
        conn
        |> put_resp_header("cache-control", "no-store")
        |> put_resp_header("pragma", "no-cache")
      end
    ```
    
    Map `{:ok, success}` to HTTP status 200 OK and `success_response` json. Map `{:error, error}` to the appropriate error status and `error_response` json.
