    ```elixir
      defmodule Success do
        @type t :: %__MODULE__{
                device_code: String.t(),
                user_code: String.t(),
                verification_uri: String.t(),
                verification_uri_complete: String.t() | nil,
                expires_in: pos_integer(),

      defmodule Error do
        @type t :: %__MODULE__{
                status: pos_integer(),
                error: String.t(),

                reason_code: atom()
              }
        defstruct [:status, :error, :error_description, :reason_code]
      end
    ```
    
    Implement `authorize(%{params: params, authorization: auth, opts: opts})`. Pipeline should:
    1. Authenticate client using `Lockspire.Protocol.ClientAuth`.
    2. Generate user and device codes via `Lockspire.Security.DeviceCode`.
    3. Issue `%Lockspire.Domain.DeviceAuthorization{}` using generated codes and client context.
    4. Persist to store using `Keyword.get(opts, :device_code_store).put_device_authorization(device_auth)`.
    5. Return `{:ok, %Success{}}` with generated plaintext codes, `verification_uri` (can be configurable or defaulted to `opts[:verification_uri] || "https://example.com/device"` for now), and `expires_in` mapped from the TTL.

    Core Protocol Pipeline Pattern:
    ```elixir
        with {:ok, %Client{} = client} <- authenticate_client(params, authorization, request),
             {:ok, validated} <- validate_request(params, client),
             {:ok, %DeviceAuthorizationState{} = device_auth} <-
               persist_device_authorization(validated, request, now) do
          {:ok,
           %Success{
             device_code: device_auth.device_code, # Use plaintext for response
             user_code: device_auth.user_code,
             verification_uri: device_auth.verification_uri,
             expires_in: DateTime.diff(device_auth.expires_at, now, :second)
           }}
        else
          {:error, %Error{} = error} ->
            {:error, error}
        end
    ```
